```rust
    let tmp = TempDir::new().expect("tempdir");
    let mut cmd = cli();
    cmd.current_dir(tmp.path())
        .arg("--output-dir").arg(tmp.path())
        .env("NO_COLOR", "1");
    (cmd, tmp)
}

pub fn fixtures_dir() -> PathBuf {
    PathBuf::from(env!("CARGO_MANIFEST_DIR")).join("tests/fixtures/domains.test")
```

```yaml
name: Release
on:
  push:

          - os: ubuntu-latest

      - uses: dtolnay/rust-toolchain@stable
        with:
          targets: ${{ matrix.target }}
      - run: cargo build --release --locked --target ${{ matrix.target }}
      - name: Package
        shell: bash
        run: |
          cd nthpartyfinder/target/${{ matrix.target }}/release
          tar czf nthpartyfinder-${{ matrix.target }}.tgz nthpartyfinder*
          shasum -a 256 nthpartyfinder-${{ matrix.target }}.tgz > nthpartyfinder-${{ matrix.target }}.tgz.sha256
      - uses: softprops/action-gh-release@v2
        with:
          files: |
            nthpartyfinder/target/${{ matrix.target }}/release/nthpartyfinder-${{ matrix.target }}.tgz
```
